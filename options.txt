usage: RelRe.jl -i IN [-o OUT] [-s START] [-e END] [-f FUTURE]
                -b BASELINE [-j [SUBJECTS...]] [--ftol_abs FTOL_ABS]
                [--ftol_rel FTOL_REL] [-l LEN] [-a ALPHA] [-t THETA]
                [-d DELTA] -w WIDTH [-D] [-q] [-g] [-c] [-u] [-h]

optional arguments:
  -i, --in IN           input file containing temporal count data of
                        variants
  -o, --out OUT         prefix of output files (default: "")
  -s, --start START     start date of the analysis (default: "")
  -e, --end END         end date of the analysis (default: "")
  -f, --future FUTURE   duration in days for predicting variant
                        frequencies (type: Int64, default: 0)
  -b, --baseline BASELINE
                        variant used as the baseline of relative
                        reproduction numbers (type: Symbol)
  -j, --subjects [SUBJECTS...]
                        list of variants to calculate relative
                        reproduction numbers (type: Symbol)
  --ftol_abs FTOL_ABS   stopping criterion used as ftol_abs in NLopt
                        (type: Float64, default: 0.0)
  --ftol_rel FTOL_REL   stopping criterion used as ftol_rel in NLopt
                        (type: Float64, default: 1.0e-8)
  -l, --len LEN         truncation point of gamma distribution for
                        generation time (type: Int64, default: -1)
  -a, --alpha ALPHA     shape parameter of gamma distribution for
                        generation time (type: Float64, default: 2.03)
  -t, --theta THETA     scale parameter of gamma distribution for
                        generation time (type: Float64, default: 2.32)
  -d, --delta DELTA     unit time of calculation (in days) (type:
                        Float64, default: 0.5)
  -w, --width WIDTH     bin width of observations: D (Daily),W
                        (Weekly),or M (Monthly) (type: Symbol,
                        default: :D)
  -D, --Dirichlet       use Dirichlet multinomial as the observation
                        model
  -q, --frequency       calculate the time course of variant
                        frequencies
  -g, --estimate_GT     estimate relative generation times of variants
  -c, --estimate_CI     estimate 95% confidence intervals
  -u, --undetected      assume all variants exist undetected from the
                        start date
  -h, --help            show this help message and exit
